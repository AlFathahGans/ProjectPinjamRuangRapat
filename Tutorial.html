Tambahkan di seeder : use Illuminate\Support\Facades\DB; 18. php artisan db:seed --class=StatusSeeder 19. php artisan db:seed --class=ApprovalStatusSeeder 20. php artisan make:controller RoomController --resource 21. php artisan make:controller BookingController --resource Install Jajra : - composer require yajra/laravel-datatables-oracle - Tambahkan Yajra\DataTables\DataTablesServiceProvider::class, di config/app.php providers - php artisan vendor:publish --provider="Yajra\DataTables\DataTablesServiceProvider"
